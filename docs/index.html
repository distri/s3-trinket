blobToS3 = (uploader, namespace, blob) ->
  deferred = $.Deferred()

  SHA1 blob, (sha) ->
    key = "#{namespace}/#{sha}"

    uploader.upload
      key: key
      blob: blob
    .then ->
      deferred.resolve key
    , (error) ->
      deferred.reject(error)

  deferred.promise()

getUserFromPolicy = (policy) ->
  conditions = JSON.parse(atob(policy.policy)).conditions.filter ([a, b, c]) ->
    a is "starts-with" and b is "$key"

  conditions[0][2]
